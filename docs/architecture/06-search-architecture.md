# Part 6: Search Architecture

AtlasMail provides instant local search (SQLite FTS5 in Electron, sql.js in
web) and comprehensive server-side search (PostgreSQL tsvector). The client
always searches locally first for speed, then optionally queries the server
for completeness.

---

## 6.1 Search Strategy Overview

```
  User types in search bar
         │
         ▼
  ┌──────────────────────┐
  │  Debounce (200ms)     │
  └──────┬───────────────┘
         │
         ▼
  ┌──────────────────────┐     ┌──────────────────────┐
  │  Local Search         │     │  Server Search        │
  │  (SQLite FTS5)        │     │  (PostgreSQL)         │
  │                       │     │                       │
  │  Response: < 10ms     │     │  Response: < 100ms    │
  │  Covers: cached data  │     │  Covers: all data     │
  └──────┬───────────────┘     └──────┬───────────────┘
         │                            │
         ▼                            ▼
  ┌──────────────────────────────────────────────────────┐
  │  Merge & deduplicate results                          │
  │  Show local results immediately, augment with server  │
  └──────────────────────────────────────────────────────┘
```

## 6.2 SQLite FTS5 Search (Local)

### Schema (defined in Part 2):

```sql
CREATE VIRTUAL TABLE emails_fts USING fts5(
    subject,
    from_name,
    from_address,
    body_text,
    content='emails',
    content_rowid='rowid',
    tokenize='porter unicode61'
);
```

### Tokenizer choice:
- `porter` — applies Porter stemming so "running" matches "run"
- `unicode61` — handles Unicode characters, case-folding, diacritics

### Query implementation:

```typescript
// packages/client/src/lib/local-db.ts (for web via sql.js)
// packages/desktop/src/services/local-db.ts (for Electron via better-sqlite3)

interface SearchOptions {
  query: string;
  accountId: string;
  category?: string;
  from?: string;
  after?: string;      // ISO date
  before?: string;     // ISO date
  hasAttachment?: boolean;
  limit?: number;
  offset?: number;
}

function searchEmailsLocal(db: Database, options: SearchOptions): SearchResult[] {
  const { query, accountId, category, from, after, before, limit = 20, offset = 0 } = options;

  // Build FTS5 query with column weights
  // FTS5 supports column filters: {subject}: term
  // and boolean operators: AND, OR, NOT
  const ftsQuery = buildFtsQuery(query);

  let sql = `
    SELECT
      e.id,
      e.thread_id,
      e.subject,
      e.from_name,
      e.from_address,
      e.internal_date,
      e.snippet,
      snippet(emails_fts, 3, '<mark>', '</mark>', '...', 40) as match_snippet,
      rank
    FROM emails_fts
    JOIN emails e ON e.rowid = emails_fts.rowid
    JOIN threads t ON t.id = e.thread_id
    WHERE emails_fts MATCH ?
      AND e.account_id = ?
  `;

  const params: any[] = [ftsQuery, accountId];

  if (category) {
    sql += ` AND t.category = ?`;
    params.push(category);
  }

  if (from) {
    sql += ` AND (e.from_address LIKE ? OR e.from_name LIKE ?)`;
    params.push(`%${from}%`, `%${from}%`);
  }

  if (after) {
    sql += ` AND e.internal_date >= ?`;
    params.push(after);
  }

  if (before) {
    sql += ` AND e.internal_date <= ?`;
    params.push(before);
  }

  sql += ` ORDER BY rank LIMIT ? OFFSET ?`;
  params.push(limit, offset);

  return db.prepare(sql).all(...params);
}

function buildFtsQuery(userQuery: string): string {
  // Parse user query into FTS5 syntax
  // Support operators: from:alice subject:meeting has:attachment
  const tokens = userQuery.trim().split(/\s+/);
  const ftsTokens: string[] = [];

  for (const token of tokens) {
    if (token.startsWith('from:')) {
      const value = token.slice(5);
      ftsTokens.push(`{from_name from_address}: ${value}`);
    } else if (token.startsWith('subject:')) {
      const value = token.slice(8);
      ftsTokens.push(`{subject}: ${value}`);
    } else {
      // General search across all columns
      // Use prefix matching for partial words: term*
      ftsTokens.push(`${token}*`);
    }
  }

  return ftsTokens.join(' AND ');
}
```

### FTS5 rank boosting:

FTS5's built-in `rank` function uses BM25. We can customize column weights:

```sql
-- Weight: subject=10, from_name=5, from_address=5, body_text=1
SELECT *, rank FROM emails_fts
WHERE emails_fts MATCH ?
ORDER BY bm25(emails_fts, 10.0, 5.0, 5.0, 1.0)
LIMIT 20;
```

This makes subject matches rank higher than body matches.

## 6.3 PostgreSQL Full-Text Search (Server)

### Schema (defined in Part 2):

```sql
-- Weighted tsvector column with auto-update trigger
search_vector TSVECTOR;

-- Generated by trigger:
NEW.search_vector :=
    setweight(to_tsvector('english', COALESCE(NEW.subject, '')), 'A') ||
    setweight(to_tsvector('english', COALESCE(NEW.from_name, '')), 'B') ||
    setweight(to_tsvector('english', COALESCE(NEW.from_address, '')), 'B') ||
    setweight(to_tsvector('english', COALESCE(NEW.body_text, '')), 'C');

-- GIN index for fast lookup
CREATE INDEX idx_emails_search ON emails USING GIN (search_vector);
```

### Query implementation:

```typescript
// packages/server/src/services/search.service.ts

import { sql } from 'drizzle-orm';

interface ServerSearchOptions {
  query: string;
  accountId: string;
  category?: string;
  from?: string;
  to?: string;
  after?: Date;
  before?: Date;
  hasAttachment?: boolean;
  inLabel?: string;
  page?: number;
  perPage?: number;
}

async function searchEmails(options: ServerSearchOptions) {
  const {
    query, accountId, category, from, to,
    after, before, hasAttachment, inLabel,
    page = 1, perPage = 20,
  } = options;

  const offset = (page - 1) * perPage;

  // Build the tsquery from user input
  // plainto_tsquery: simple words joined with AND
  // phraseto_tsquery: exact phrase matching
  // websearch_to_tsquery: supports "quoted phrases", OR, -exclusions
  const tsqueryType = query.includes('"')
    ? 'websearch_to_tsquery'
    : 'plainto_tsquery';

  let whereClause = sql`
    e.account_id = ${accountId}
    AND e.search_vector @@ ${sql.raw(tsqueryType)}('english', ${query})
  `;

  if (category) {
    whereClause = sql`${whereClause} AND t.category = ${category}`;
  }

  if (from) {
    whereClause = sql`${whereClause} AND (
      e.from_address ILIKE ${'%' + from + '%'}
      OR e.from_name ILIKE ${'%' + from + '%'}
    )`;
  }

  if (to) {
    whereClause = sql`${whereClause} AND e.to_addresses::text ILIKE ${'%' + to + '%'}`;
  }

  if (after) {
    whereClause = sql`${whereClause} AND e.internal_date >= ${after}`;
  }

  if (before) {
    whereClause = sql`${whereClause} AND e.internal_date <= ${before}`;
  }

  if (hasAttachment) {
    whereClause = sql`${whereClause} AND t.has_attachments = true`;
  }

  // Main search query
  const results = await db.execute(sql`
    SELECT
      e.id as email_id,
      e.thread_id,
      e.subject,
      e.from_name,
      e.from_address,
      e.internal_date,
      ts_headline(
        'english',
        COALESCE(e.body_text, ''),
        ${sql.raw(tsqueryType)}('english', ${query}),
        'StartSel=<mark>, StopSel=</mark>, MaxWords=40, MinWords=20, MaxFragments=2'
      ) as snippet,
      ts_rank_cd(
        e.search_vector,
        ${sql.raw(tsqueryType)}('english', ${query}),
        32  -- normalization: rank / (1 + rank)
      ) as rank
    FROM emails e
    JOIN threads t ON t.id = e.thread_id
    WHERE ${whereClause}
    ORDER BY rank DESC, e.internal_date DESC
    LIMIT ${perPage}
    OFFSET ${offset}
  `);

  // Count query (for pagination)
  const countResult = await db.execute(sql`
    SELECT COUNT(*) as total
    FROM emails e
    JOIN threads t ON t.id = e.thread_id
    WHERE ${whereClause}
  `);

  return {
    results: results.rows,
    total: Number(countResult.rows[0].total),
    page,
    perPage,
  };
}
```

### Search query syntax (user-facing):

| Syntax              | Example                  | Meaning                       |
|---------------------|--------------------------|-------------------------------|
| Plain words         | `project update`         | Emails containing both words  |
| Quoted phrase        | `"project update"`       | Exact phrase match            |
| from:               | `from:alice`             | Sender contains "alice"       |
| to:                 | `to:bob@example.com`     | Recipient matches             |
| subject:            | `subject:meeting`        | Subject contains "meeting"    |
| has:attachment       | `has:attachment`         | Has attachments               |
| in:                 | `in:sent`                | In a specific folder          |
| after:              | `after:2026-01-01`       | After a date                  |
| before:             | `before:2026-02-01`      | Before a date                 |
| -                   | `meeting -cancelled`     | Exclude "cancelled"           |
| OR                  | `alice OR bob`           | Either term                   |

The query parser in `packages/shared/src/utils/search-parser.ts` transforms
user syntax into both FTS5 and PostgreSQL tsquery formats.

## 6.4 Client-Side Search Flow

```typescript
// packages/client/src/hooks/use-search.ts

import { useQuery } from '@tanstack/react-query';
import { useDebouncedValue } from './use-debounced-value';
import { useLocalDb } from './use-local-db';
import { apiClient } from '../lib/api-client';

export function useSearch(query: string) {
  const debouncedQuery = useDebouncedValue(query, 200);
  const db = useLocalDb();

  // Local search — instant results
  const localResults = useQuery({
    queryKey: ['search', 'local', debouncedQuery],
    queryFn: () => db.searchEmails(debouncedQuery),
    enabled: debouncedQuery.length >= 2,
    staleTime: 0,  // Always re-run for fresh results
  });

  // Server search — comprehensive results (only in web mode)
  const serverResults = useQuery({
    queryKey: ['search', 'server', debouncedQuery],
    queryFn: () => apiClient.get('/search', { params: { q: debouncedQuery } }),
    enabled: debouncedQuery.length >= 2,
    staleTime: 30_000,
  });

  // Merge results: local first, then server results not in local set
  const merged = mergeSearchResults(
    localResults.data ?? [],
    serverResults.data?.data ?? []
  );

  return {
    results: merged,
    isLoading: localResults.isLoading,
    isLoadingServer: serverResults.isLoading,
  };
}

function mergeSearchResults(local: SearchResult[], server: SearchResult[]): SearchResult[] {
  const seenThreadIds = new Set(local.map(r => r.thread_id));
  const serverOnly = server.filter(r => !seenThreadIds.has(r.thread_id));

  return [
    ...local,
    ...serverOnly.map(r => ({ ...r, source: 'server' as const })),
  ];
}
```

## 6.5 Search Performance Targets

| Metric                           | Target      |
|----------------------------------|-------------|
| Local FTS5 query (10k emails)    | < 5ms       |
| Local FTS5 query (100k emails)   | < 20ms      |
| PostgreSQL search (1M emails)    | < 100ms     |
| Debounce delay                   | 200ms       |
| First results visible            | < 250ms     |

### Index maintenance:
- FTS5 indexes are automatically maintained via triggers.
- PostgreSQL GIN indexes are automatically maintained.
- Run `REINDEX` on PostgreSQL during off-peak hours weekly (optional).
- FTS5 `optimize` command can be run in Electron during idle periods.
